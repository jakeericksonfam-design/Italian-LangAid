import streamlit as st
import google.generativeai as genai
import os

# --- CONFIGURATION ---
# Ideally, store this in a secrets file, but for local testing, you can input it in the sidebar
st.set_page_config(page_title="Italiano AI Tutor", page_icon="🇮🇹")

# --- SIDEBAR SETTINGS ---
with st.sidebar:
    st.title("⚙️ Settings")
    api_key = st.text_input("Enter Google API Key", type="password")
    
    st.divider()
    
    level = st.selectbox("Student Level", ["A1 (Beginner)", "A2 (Elementary)", "B1 (Intermediate)"])
    response_lang = st.radio("Explain in:", ["English", "Italian"])
    
    st.divider()
    st.markdown("### 🛠️ Tools")
    if st.button("📝 Generate Vocabulary Quiz"):
        st.session_state.trigger_quiz = True
    if st.button("📖 Explain Grammar Topic"):
        st.session_state.trigger_grammar = True
    if st.button("🗑️ Clear Chat"):
        st.session_state.messages = []
        st.session_state.trigger_quiz = False
        st.session_state.trigger_grammar = False
        st.rerun()

# --- AI CONFIGURATION ---
if api_key:
    genai.configure(api_key=api_key)
    # Using Gemini 1.5 Flash (Fast and Free Tier eligible)
    model = genai.GenerativeModel('gemini-1.5-flash')
else:
    st.warning("Please enter your API Key in the sidebar to start.")
    st.stop()

# --- SYSTEM PROMPT (THE TEACHER) ---
# This is the most important part. It defines how the AI behaves.
system_instruction = f"""
You are an expert Italian Language Tutor. 
Your student is at the {level} level.
Your primary goal is to help them learn Italian through conversation, exercises, and explanations.

RULES:
1. **Language:** Speak primarily in {response_lang}. If the user asks a question in Italian, answer in Italian but provide an English translation if the concept is complex.
2. **Level Appropriateness:** Do not use complex grammar or vocabulary above the {level} level unless you are explicitly teaching it.
3. **Corrections:** If the user makes a mistake, gently correct them. Show the incorrect sentence, the correct version, and a brief explanation in {response_lang}.
4. **Encouragement:** Be encouraging and patient.
5. **Exercises:** When asked for exercises, provide 5 questions. Wait for the user to answer before providing the solution key.
6. **Grammar:** When explaining grammar, use clear tables or bullet points.
"""

# --- SESSION STATE MANAGEMENT ---
if "messages" not in st.session_state:
    st.session_state.messages = []

if "trigger_quiz" not in st.session_state:
    st.session_state.trigger_quiz = False

if "trigger_grammar" not in st.session_state:
    st.session_state.trigger_grammar = False

# --- HANDLE SPECIAL TOOL TRIGGERS ---
user_prompt = None

if st.session_state.trigger_quiz:
    user_prompt = "Please generate a 5-question vocabulary quiz based on common travel phrases suitable for my level. Do not show the answers yet."
    st.session_state.trigger_quiz = False # Reset trigger
    st.rerun()

if st.session_state.trigger_grammar:
    user_prompt = "Please ask me what specific grammar topic I am struggling with (e.g., articles, verb conjugations, prepositions), then explain it."
    st.session_state.trigger_grammar = False # Reset trigger
    st.rerun()

# --- CHAT INTERFACE ---
st.title("🇮🇹 Italiano AI Tutor")
st.caption(f"Current Level: {level} | Explanations: {response_lang}")

# Display chat history
for message in st.session_state.messages:
    with st.chat_message(message["role"]):
        st.markdown(message["content"])

# User Input
if prompt := st.chat_input("Ask a question or practice Italian..."):
    # Use the tool prompt if triggered, otherwise use user input
    final_prompt = prompt if not user_prompt else user_prompt
    
    # Add user message to history
    st.session_state.messages.append({"role": "user", "content": final_prompt})
    with st.chat_message("user"):
        st.markdown(final_prompt)

    # Generate Response
    with st.chat_message("assistant"):
        message_placeholder = st.empty()
        message_placeholder.markdown("Thinking...")
        
        try:
            # Create a chat session with history
            chat = model.start_chat(history=[])
            
            # Send the system instruction + current prompt
            # Note: Gemini handles system instructions via generation_config or prepended text
            full_context = f"{system_instruction}\n\nUser: {final_prompt}"
            
            response = model.generate_content(full_context)
            ai_response = response.text
            
            message_placeholder.markdown(ai_response)
            st.session_state.messages.append({"role": "assistant", "content": ai_response})
        except Exception as e:
            st.error(f"An error occurred: {e}")
